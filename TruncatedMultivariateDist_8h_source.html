146 template<class Dist>
147 double TruncatedMultivariateDist<Dist>::compute_truncated_pdf_integral(const NdimVecT &lbound, const NdimVecT &ubound, double lbound_cdf) const
148 {
149  const IdxT N = Dist::num_dim();
150  if(lbound_cdf==0 && arma::all(lbound==-INFINITY)) return this->Dist::cdf(ubound);
151  double pdf_integral = (N%2==0) ? lbound_cdf : -lbound_cdf; //account for the lbound() vertex here.
152 
153  //n iterates through all integers less than 2^N. we use the binary repr of N to choose ubound or lbound
154  // 0 bit = use ubound(k)
155  // 1 bit = use lbound(k)
156  // all 1's is skipped as we account for that in initial pdf_integral value
157  for(IdxT n=0; n<pow(2,N)-1; n++) {
158  IdxT b=n; //b is used as a binary repr.
159  IdxT k=0; //k is index we are considering
160  IdxT flips=0;
161  NdimVecT v = ubound;
162 // std::cout<<"N:"<<N<<" n:"<<n<<" ["<<std::bitset<8>(n)<<"]\n";
163  while(b && k<N) {
164 // std::cout<<"k:"<<k<<" b:"<<std::bitset<8>(b)<<"\n";
165  if(b&0x1) {
166  v(k) = lbound(k);
167  flips++;
168  }
169  k++;
170  b>>=1;
171  }
172 // std::cout<<"Ubound:"<<ubound.t();
173 // std::cout<<" v:"<<v.t();
174 // std::cout<<"Lbound:"<<lbound.t();
175  double cdf_val = arma::any(v==-INFINITY) ? 0 : this->Dist::cdf(v);
176 // std::cout<<" cdf:"<<cdf_val<<"\n";
177 // std::cout<<" flips:"<<flips<<"\n";
178  if(flips%2==1) cdf_val = -cdf_val; //odd number of flips
179  pdf_integral += cdf_val;
180  }
181  return pdf_integral;
182 }

294 template<class RngT>
295 typename TruncatedMultivariateDist<Dist>::NdimVecT
296 TruncatedMultivariateDist<Dist>::mcmc_sample(RngT &rng) const
297 {
298  const int burnin=10;
299  const int keep_every =10;
300  auto& lb = this->lbound();
301  auto& ub = this->ubound();
302  std::uniform_real_distribution<double> uniform;
303 
304  this->mcmc.mutex.lock();
305  NdimVecT sample = mcmc.sample;
306  double sample_rllh = mcmc.rllh;
307  int N = mcmc.nsample;
308  this->mcmc.mutex.unlock();
309  if(N==0) {
310  sample = .5*(lb+ub);
311  sample_rllh = this->rllh(sample);
312  }
313  do {
314  N++;
315  NdimVecT can_sample;
316  for(IdxT k=0;k<this->num_dim();k++) can_sample(k) = uniform(rng)*(ub(k)-lb(k)) + lb(k);
317  double can_rllh = this->rllh(can_sample);
318  double alpha = std::min(1., exp(can_rllh - sample_rllh));
319 // std::cout<<"N:"<<N<<" alpha:"<<alpha<<" rllh: "<<can_rllh<<" can_sample "<< can_sample.t();
320  if(uniform(rng) < alpha) {
321  //Accept
322 // std::cout<<"Accept!\n";
323  sample = can_sample;
324  sample_rllh = can_rllh;
325  }
326  } while(N<=burnin || N%keep_every != 0);
327 
328  this->mcmc.mutex.lock();
329  mcmc.sample = sample;
330  mcmc.rllh = sample_rllh;
331  mcmc.nsample = N;
332  this->mcmc.mutex.unlock();
333 
334  return sample;
335 }
