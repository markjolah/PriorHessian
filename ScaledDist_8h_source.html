76 void ScaledDist<Dist>::set_bounds(double lbound, double ubound)
77 {
78  if( !(lbound < ubound) || !std::isfinite(lbound) || !std::isfinite(ubound) ){
79  std::ostringstream msg;
80  msg<<"set_bounds: Invalid bounds lbound:"<<lbound<<" should be < ubound:"<<ubound;
81  throw ParameterValueError(msg.str());
82  }
83  double unscaled_delta = unscaled_ubound() - unscaled_lbound();
84  if( !std::isfinite(unscaled_delta) ){
85  std::ostringstream msg;
86  msg<<"set_bounds: Invalid distribution with non-finite bounds cannot be scaled."
87  <<"Unscaled: [ lbound:"<<unscaled_lbound()<<" ubound:"<<unscaled_ubound()<<"]";
88  throw ParameterValueError(msg.str());
89  }
90 
91  scaling_ratio = (ubound - lbound) / unscaled_delta;
92  llh_scaling_const = -log(scaling_ratio);
93  _scaled_lbound = lbound;
94  _scaled_ubound = ubound;
95 }
