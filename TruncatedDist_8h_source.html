81 template<class Dist>
82 void TruncatedDist<Dist>::set_bounds(double lbound, double ubound)
83 {
84  if( !(lbound >= global_lbound()) ) { //This form of comparison handles NaNs
85  std::ostringstream msg;
86  msg<<"set_bounds: Invalid lbound:"<<lbound<<" with regard to global_lbound:"<<global_lbound();
87  throw ParameterValueError(msg.str());
88  }
89  if( !(ubound <= global_ubound()) ) { //This form of comparison handles NaNs
90  std::ostringstream msg;
91  msg<<"set_bounds: Invalid ubound:"<<ubound<<" with regard to global_ubound:"<<global_ubound();
92  throw ParameterValueError(msg.str());
93  }
94  if( !(lbound < ubound) ) {
95  std::ostringstream msg;
96  msg<<"set_bounds: Invalid bounds lbound:"<<lbound<<" >= ubound:"<<ubound;
97  throw ParameterValueError(msg.str());
98  }
99  bool truncated = (lbound>global_lbound() || ubound<global_ubound());
100  if(truncated) {
101  lbound_cdf = (lbound==global_lbound()) ? 0 : Dist::cdf(lbound);
102  double ubound_cdf = (ubound==global_ubound()) ? 1 : Dist::cdf(ubound);
103  bounds_pdf_integral = ubound_cdf - lbound_cdf;
104  if(bounds_pdf_integral<min_bounds_pdf_integral) {
105  std::ostringstream msg;
106  msg<<"TruncatedDist::set_bounds: bounds:["<<lbound<<","<<ubound<<"] with cdf:["<<lbound_cdf<<","<<ubound_cdf
107  <<"] have delta: "<<bounds_pdf_integral<<" < min_delta = "<<min_bounds_pdf_integral
108  <<". Bounds cover too small a portion of the domain for accuarate truncation.";
109  throw ParameterValueError(msg.str());
110  }
111  llh_truncation_const = -log(bounds_pdf_integral);
112  } else {
113  lbound_cdf = 0;
114  bounds_pdf_integral = 1;
115  llh_truncation_const = 0;
116  }
117  _truncated = truncated;
118  _truncated_lbound = lbound;
119  _truncated_ubound = ubound;
120 }
