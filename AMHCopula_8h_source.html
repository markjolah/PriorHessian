245 template<int Ndim>
246 template<class Vec>
247 typename AMHCopula<Ndim>::NdimMatT
248 AMHCopula<Ndim>::hess(const Vec &u) const
249 {
250  NdimMatT hess;
251  auto gterms = compute_d2_gen_terms_only(theta(),u);
252  NdimVecT d1_igen_u;
253  for(int i=0; i<Ndim; i++) {
254  auto igterms = compute_d2_igen_terms_only(theta(),u(i));
255  hess(i,i) = square(igterms.d1_igen_ui)*gterms.xi_n_t + igterms.d2_igen_ui*gterms.eta_n_np1_t + igterms.ixi_1_ui;
256  d1_igen_u(i) = igterms.d1_igen_ui;
257  for(int j=0; j<i; j++) hess(j,i) = d1_igen_u(i)*d1_igen_u(j)*gterms.xi_n_t;
258  }
259  return arma::symmatu(hess);

305 double AMHCopula<Ndim>::compute_z(double theta, const Vec &u)
306 {
307  double z = theta;

320 template<int Ndim>
321 template<class Vec>
322 typename AMHCopula<Ndim>::PDFTerms
323 AMHCopula<Ndim>::compute_pdf_terms(double theta, const Vec &u)
324 {
325  PDFTerms terms;
326  terms.z = theta;
327  terms.prod_d1_igen = 1;
328  double one_m_theta = 1-theta;
329  for(IdxT i=0; i<Ndim; i++) {
330  double W_inv = 1/(u(i)*theta+one_m_theta);
331  terms.z *= W_inv*u(i);
332  terms.prod_d1_igen *= one_m_theta*W_inv/u(i);
333  }
334  return terms;
335 }
