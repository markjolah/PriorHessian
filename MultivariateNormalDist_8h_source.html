137  double compute_quadratic_from_symmetric(IdxT Ndim, const Vec &v, const Mat &A)
138  {
139  double z=0;
140  for(IdxT c=0; c<Ndim; c++) {
141  double vc = v(c);
142  for(IdxT r=0; r<c; r++) z+= 2*vc*v(r)*A(r,c); //Account for both off-diagonal elements simultaneously.
143  z+=square(vc)*A(c,c);
144  }
145  return z;

241 {

243  if(Z != num_params()-Ndim) {
244  std::ostringstream msg;

251  for(IdxT j=0;j<Ndim-1;j++) for(IdxT i=j+1;i<Ndim; i++) m(i,j) = m(j,i); //copy upper triangle to lower.

397 template<IdxT Ndim>
398 template<class Mat>
399 void MultivariateNormalDist<Ndim>::set_sigma(Mat&& val)
400 {
401  //Mat is upper-triangular symmetric, positive definite.
402 // std::cout<<"sigma: "<<val<<std::endl;
403  if(!val.is_finite()) throw ParameterValueError("Sigma matrix is not-finite.");
404  if(arma::any(val.diag()<=0)) throw ParameterValueError("Sigma matrix is not positive definite.");
405  try{
406  _sigma_chol = arma::chol(arma::symmatu(val),"lower");
407  } catch (std::runtime_error &e) {
408  throw ParameterValueError("Cholesky decomposition failure. Sigma is not positive definite.");
409  }
410  try {
411  _sigma_inv = arma::symmatu(arma::inv_sympd(arma::symmatu(val)));
412  } catch (std::logic_error &e) {
413  std::ostringstream msg;
414  msg<<"Bad sigma size: "<<val.n_rows<<","<<val.n_cols<<"\n";
415  throw ParameterSizeError(msg.str());
416  } catch (std::runtime_error &e) {
417  throw ParameterValueError("Sigma is not symmetric positive semi-definite with bounded eigenvalues. Numerical inversion failure.");
418  }
419  _sigma = arma::symmatu(std::forward<Mat>(val));
420  llh_const_initialized = false;
421 }
